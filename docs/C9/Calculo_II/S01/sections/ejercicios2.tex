\section{Pregunta 2}
\noindent Evalúe la integral y evalúe si conger o diverge

\[
  \int_2^{+\infty} 3e{-2x} \, dx
\]

Sug. Use la regla L'Hopital.

\subsection{Solución}

\begin{gather*}
  \int_2^{+\infty} 3e^{-2x} \, dx \\
  \lim_{b \to +\infty} \int_2^b 3e^{-2x} \, dx \\
  \int_2^{+\infty} \, dx \\
  \left[ \frac{3e^{-}}{} \right]
\end{gather*}
