\section*{Ejercicio 4}
\noindent\textbf{Dada las siguientes coordenadas cartesianas, determine sus coordenadas polares y grafíquelas en el plano polar}

\subsection*{4.1}
\[
  (-3.5 \sqrt{3}; -3.5)
\]


\begin{center}
  \begin{tikzpicture}
  \begin{polaraxis}[
    title=A polar axis,
    xtick={0,30,60,90,120,150,180,210,240,270,300,330},
    xticklabels={
      $0$, $\frac{\pi}{6}$, $\frac{\pi}{3}$,
      $\frac{\pi}{2}$, $\frac{2\pi}{3}$, $\frac{5\pi}{6}$,
      $\pi$, $\frac{7\pi}{6}$, $\frac{4\pi}{3}$,
      $\frac{3\pi}{2}$, $\frac{5\pi}{3}$, $\frac{11\pi}{6}$
    },
    ytick={0,1,2,3,4,5,6,7},
    clip=false,
  ]
    % Point P in polar coordinates: (r=7, θ=210° or 7π/6)
    \addplot+[only marks, mark=*, color=black] coordinates {(210,7)}
      node [pin=120: {\footnotesize $P\left(7;\frac{7\pi}{6}\right)$}] {};

    % Triangle from origin to point P and back to x-axis
    \addplot+ [data cs=cart, mark=none] coordinates {
      (0,0) (-3.5*sqrt(3),0) (-3.5*sqrt(3),-3.5) (0,0)
    };

    \addplot[thick, blue, domain=0:180, samples=30] (x,1.2);
    \node[above, blue, font=\footnotesize] at (axis cs:90,1.2) {$180^{\circ}$};

    % Labels for the sides of the triangle
    \node[above, font=\footnotesize] at (axis cs:180,3.5) {$3.5\sqrt{3}$};
    \node[left, font=\footnotesize] at (axis cs:195,5) {$3.5$};
    \node[below, font=\footnotesize] at (axis cs:210,4) {$7$};

    % Angle at origin: 30° (arc centered at origin)
    \addplot[thick, red, domain=180:210, samples=30] (\x,1.2);
    \node[red, font=\footnotesize] at (axis cs:195,1.9) {$30^{\circ}$};

    % Right angle at (-3.5√3, 0): 90°
    % Arc centered at (-3.5√3, 0), from 270° to 360°
    \addplot[thick, red, data cs=cart, domain=270:360, samples=30] 
      ({-3.5*sqrt(3)+0.5*cos(x)}, {0.5*sin(x)});
    \node[red, font=\footnotesize] at (axis cs:185,5) {$90^{\circ}$};

    % Angle at P: 60°
    % Arc centered at (-3.5√3, -3.5), from 30° to 90°
    \addplot[thick, red, data cs=cart, domain=30:90, samples=30] 
      ({-3.5*sqrt(3)+0.8*cos(x)}, {-3.5+0.8*sin(x)});
    \node[red, font=\footnotesize] at (axis cs:205,5.8) {$60^{\circ}$};

  \end{polaraxis}
  \end{tikzpicture} 
\end{center}
