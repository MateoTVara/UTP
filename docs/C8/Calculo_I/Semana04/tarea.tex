\begin{center}
\begin{minipage}{0.48\textwidth}
\[
\begin{aligned}
  f(x) &= 0.5(x-5)^2 + 3, \quad x < 5 \\
  y &= 0.5(x-5)^2 + 3 \\
  y - 3 &= 0.5(x-5)^2 \\

\end{aligned}
